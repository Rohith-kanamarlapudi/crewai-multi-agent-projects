```python
# Python program to calculate sum of first 10,000 terms of series
# 1 - 1/3 + 1/5 - 1/7 + ... and multiply the total by 4

def calculate_pi_approx(terms=10000):
    total = 0.0
    for i in range(terms):
        denominator = 2 * i + 1          # generates 1, 3, 5, 7, ...
        # add term with alternating sign: + for even i, - for odd i
        total += (1.0 / denominator) if i % 2 == 0 else (-1.0 / denominator)
    return total * 4                     # multiply sum by 4

if __name__ == "__main__":
    result = calculate_pi_approx(10000)
    print(result)
```
